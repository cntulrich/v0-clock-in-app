import React, { useState } from 'react';
import { Upload, FileText, Check, X, AlertCircle, Users, Download } from 'lucide-react';

export default function PDFEmployeeImport() {
  const [file, setFile] = useState(null);
  const [loading, setLoading] = useState(false);
  const [employees, setEmployees] = useState([]);
  const [error, setError] = useState('');
  const [success, setSuccess] = useState(false);
  const [stats, setStats] = useState({ total: 0, office: 0, wfh: 0, hybrid: 0 });

  const handleFileChange = async (e) => {
    const selectedFile = e.target.files[0];
    
    if (!selectedFile) return;
    
    if (selectedFile.type !== 'application/pdf') {
      setError('Please upload a PDF file');
      return;
    }
    
    setFile(selectedFile);
    setError('');
    setSuccess(false);
    await processPDF(selectedFile);
  };

  const processPDF = async (pdfFile) => {
    setLoading(true);
    setError('');
    
    try {
      const text = await window.fs.readFile(pdfFile.name, { encoding: 'utf8' });
      
      // Parse the text content
      const lines = text.split('\n');
      const parsedEmployees = [];
      let currentManager = '';
      
      for (let i = 0; i < lines.length; i++) {
        const line = lines[i].trim();
        
        // Skip empty lines and headers
        if (!line || line.includes('TEAMS COMPANY') || line.includes('Total Agents:')) {
          continue;
        }
        
        // Parse each employee line
        // Format: Manager Company AgentName Location
        const parts = line.split(/\s{2,}/); // Split by 2 or more spaces
        
        if (parts.length >= 3) {
          const manager = parts[0];
          const company = parts[1];
          const agentName = parts[2];
          const location = parts[3] || 'OFFICE';
          
          // Update current manager if this is a new section
          if (manager && manager !== currentManager) {
            currentManager = manager;
          }
          
          parsedEmployees.push({
            id: Date.now() + Math.random(),
            manager: currentManager,
            company: company,
            name: agentName,
            location: location.toUpperCase(),
            email: generateEmail(agentName),
            employeeId: `EMP${String(parsedEmployees.length + 1).padStart(4, '0')}`
          });
        }
      }
      
      // Calculate stats
      const officeCount = parsedEmployees.filter(e => e.location === 'OFFICE').length;
      const wfhCount = parsedEmployees.filter(e => e.location === 'WFH').length;
      const hybridCount = parsedEmployees.filter(e => e.location === 'HYBRID').length;
      
      setStats({
        total: parsedEmployees.length,
        office: officeCount,
        wfh: wfhCount,
        hybrid: hybridCount
      });
      
      setEmployees(parsedEmployees);
      setLoading(false);
      setSuccess(true);
      
    } catch (err) {
      setError('Failed to process PDF. Please make sure the file is uploaded correctly.');
      setLoading(false);
      console.error(err);
    }
  };

  const generateEmail = (name) => {
    // Generate email from name (firstname.lastname@company.com)
    const nameParts = name.toLowerCase().split(' ');
    if (nameParts.length >= 2) {
      return `${nameParts[0]}.${nameParts[nameParts.length - 1]}@company.com`;
    }
    return `${name.toLowerCase().replace(/\s/g, '.')}@company.com`;
  };

  const handleImport = () => {
    // Here you would save the employees to your database
    console.log('Importing employees:', employees);
    alert(`Successfully imported ${employees.length} employees!`);
    
    // Reset form
    setFile(null);
    setEmployees([]);
    setSuccess(false);
    setStats({ total: 0, office: 0, wfh: 0, hybrid: 0 });
  };

  const removeEmployee = (id) => {
    const updatedEmployees = employees.filter(emp => emp.id !== id);
    setEmployees(updatedEmployees);
    
    // Recalculate stats
    const officeCount = updatedEmployees.filter(e => e.location === 'OFFICE').length;
    const wfhCount = updatedEmployees.filter(e => e.location === 'WFH').length;
    const hybridCount = updatedEmployees.filter(e => e.location === 'HYBRID').length;
    
    setStats({
      total: updatedEmployees.length,
      office: officeCount,
      wfh: wfhCount,
      hybrid: hybridCount
    });
  };

  const downloadTemplate = () => {
    const template = `TEAMS COMPANY Agent Name Location
Manager VGI John Doe OFFICE
Manager VGI Jane Smith WFH
Manager Priority One Bob Johnson HYBRID`;
    
    const blob = new Blob([template], { type: 'text/plain' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'employee-import-template.txt';
    a.click();
    URL.revokeObjectURL(url);
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100 p-8">
      <div className="max-w-6xl mx-auto">
        <div className="bg-white rounded-lg shadow-xl p-8">
          <div className="flex items-center justify-between mb-6">
            <div>
              <h1 className="text-3xl font-bold text-gray-800 mb-2">Import Employees from PDF</h1>
              <p className="text-gray-600">Upload your employee roster PDF</p>
            </div>
            <button
              onClick={downloadTemplate}
              className="flex items-center gap-2 px-4 py-2 bg-gray-100 text-gray-700 rounded-lg hover:bg-gray-200 transition-colors"
            >
              <Download className="w-4 h-4" />
              Download Template
            </button>
          </div>

          {/* Upload Area */}
          <div className="mb-8">
            <label 
              htmlFor="pdf-upload" 
              className="flex flex-col items-center justify-center w-full h-64 border-2 border-dashed border-gray-300 rounded-lg cursor-pointer bg-gray-50 hover:bg-gray-100 transition-colors"
            >
              <div className="flex flex-col items-center justify-center pt-5 pb-6">
                <Upload className="w-12 h-12 text-gray-400 mb-4" />
                <p className="mb-2 text-sm text-gray-500">
                  <span className="font-semibold">Click to upload</span> or drag and drop
                </p>
                <p className="text-xs text-gray-500">PDF files only</p>
                {file && (
                  <div className="mt-4 flex items-center gap-2 text-sm text-blue-600">
                    <FileText className="w-4 h-4" />
                    <span>{file.name}</span>
                  </div>
                )}
              </div>
              <input 
                id="pdf-upload" 
                type="file" 
                className="hidden" 
                accept=".pdf"
                onChange={handleFileChange}
              />
            </label>
          </div>

          {/* Error Message */}
          {error && (
            <div className="mb-6 p-4 bg-red-50 border border-red-200 rounded-lg flex items-center gap-2 text-red-700">
              <AlertCircle className="w-5 h-5" />
              <span>{error}</span>
            </div>
          )}

          {/* Loading State */}
          {loading && (
            <div className="mb-6 p-4 bg-blue-50 border border-blue-200 rounded-lg">
              <div className="flex items-center gap-3">
                <div className="animate-spin rounded-full h-5 w-5 border-b-2 border-blue-600"></div>
                <span className="text-blue-700">Processing PDF...</span>
              </div>
            </div>
          )}

          {/* Success Message & Stats */}
          {success && employees.length > 0 && (
            <div className="mb-6">
              <div className="p-4 bg-green-50 border border-green-200 rounded-lg flex items-center gap-2 text-green-700 mb-4">
                <Check className="w-5 h-5" />
                <span>Found {stats.total} employees in the PDF. Review and import below.</span>
              </div>
              
              {/* Stats Cards */}
              <div className="grid grid-cols-4 gap-4 mb-6">
                <div className="bg-blue-50 p-4 rounded-lg border border-blue-200">
                  <div className="flex items-center gap-2 mb-2">
                    <Users className="w-5 h-5 text-blue-600" />
                    <span className="text-sm font-medium text-blue-900">Total</span>
                  </div>
                  <p className="text-2xl font-bold text-blue-600">{stats.total}</p>
                </div>
                
                <div className="bg-purple-50 p-4 rounded-lg border border-purple-200">
                  <div className="flex items-center gap-2 mb-2">
                    <span className="text-sm font-medium text-purple-900">Office</span>
                  </div>
                  <p className="text-2xl font-bold text-purple-600">{stats.office}</p>
                </div>
                
                <div className="bg-green-50 p-4 rounded-lg border border-green-200">
                  <div className="flex items-center gap-2 mb-2">
                    <span className="text-sm font-medium text-green-900">WFH</span>
                  </div>
                  <p className="text-2xl font-bold text-green-600">{stats.wfh}</p>
                </div>
                
                <div className="bg-orange-50 p-4 rounded-lg border border-orange-200">
                  <div className="flex items-center gap-2 mb-2">
                    <span className="text-sm font-medium text-orange-900">Hybrid</span>
                  </div>
                  <p className="text-2xl font-bold text-orange-600">{stats.hybrid}</p>
                </div>
              </div>
            </div>
          )}

          {/* Parsed Employees Table */}
          {employees.length > 0 && (
            <div className="mb-6">
              <h2 className="text-xl font-semibold text-gray-800 mb-4">Employees to Import</h2>
              <div className="border border-gray-200 rounded-lg overflow-hidden max-h-96 overflow-y-auto">
                <table className="min-w-full divide-y divide-gray-200">
                  <thead className="bg-gray-50 sticky top-0">
                    <tr>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Manager</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Company</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Name</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Email</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Location</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">ID</th>
                      <th className="px-4 py-3 text-right text-xs font-medium text-gray-500 uppercase tracking-wider">Action</th>
                    </tr>
                  </thead>
                  <tbody className="bg-white divide-y divide-gray-200">
                    {employees.map((employee) => (
                      <tr key={employee.id} className="hover:bg-gray-50">
                        <td className="px-4 py-3 whitespace-nowrap text-sm text-gray-900">{employee.manager}</td>
                        <td className="px-4 py-3 whitespace-nowrap text-sm text-gray-500">{employee.company}</td>
                        <td className="px-4 py-3 whitespace-nowrap text-sm font-medium text-gray-900">{employee.name}</td>
                        <td className="px-4 py-3 whitespace-nowrap text-sm text-gray-500">{employee.email}</td>
                        <td className="px-4 py-3 whitespace-nowrap">
                          <span className={`px-2 py-1 text-xs font-semibold rounded-full ${
                            employee.location === 'OFFICE' ? 'bg-purple-100 text-purple-800' :
                            employee.location === 'WFH' ? 'bg-green-100 text-green-800' :
                            'bg-orange-100 text-orange-800'
                          }`}>
                            {employee.location}
                          </span>
                        </td>
                        <td className="px-4 py-3 whitespace-nowrap text-sm text-gray-500">{employee.employeeId}</td>
                        <td className="px-4 py-3 whitespace-nowrap text-right text-sm font-medium">
                          <button
                            onClick={() => removeEmployee(employee.id)}
                            className="text-red-600 hover:text-red-900 transition-colors"
                            title="Remove employee"
                          >
                            <X className="w-5 h-5" />
                          </button>
                        </td>
                      </tr>
                    ))}
                  </tbody>
                </table>
              </div>

              {/* Import Button */}
              <div className="mt-6 flex justify-end gap-3">
                <button
                  onClick={() => {
                    setFile(null);
                    setEmployees([]);
                    setSuccess(false);
                  }}
                  className="px-6 py-3 bg-gray-200 text-gray-700 font-semibold rounded-lg hover:bg-gray-300 transition-colors"
                >
                  Cancel
                </button>
                <button
                  onClick={handleImport}
                  className="px-6 py-3 bg-blue-600 text-white font-semibold rounded-lg hover:bg-blue-700 transition-colors shadow-md hover:shadow-lg flex items-center gap-2"
                >
                  <Check className="w-5 h-5" />
                  Import {employees.length} Employee{employees.length !== 1 ? 's' : ''}
                </button>
              </div>
            </div>
          )}

          {/* Instructions */}
          <div className="mt-8 p-6 bg-gray-50 rounded-lg border border-gray-200">
            <h3 className="font-semibold text-gray-800 mb-3">Expected PDF Format:</h3>
            <div className="bg-white p-4 rounded border border-gray-300 mb-4 font-mono text-sm">
              <div>TEAMS COMPANY Agent Name Location</div>
              <div>Anthony VGI Alison Torrez OFFICE</div>
              <div>Anthony Priority One Betty Aragundy OFFICE</div>
              <div>Ashley VGI Anthony Martinez WFH</div>
            </div>
            <ul className="space-y-2 text-sm text-gray-600">
              <li className="flex items-start gap-2">
                <span className="text-blue-600 mt-1">•</span>
                <span>Columns: Manager/Team, Company, Agent Name, Location</span>
              </li>
              <li className="flex items-start gap-2">
                <span className="text-blue-600 mt-1">•</span>
                <span>Locations: OFFICE, WFH, or HYBRID</span>
              </li>
              <li className="flex items-start gap-2">
                <span className="text-blue-600 mt-1">•</span>
                <span>Emails will be auto-generated from names</span>
              </li>
            </ul>
          </div>
        </div>
      </div>
    </div>
  );
}
